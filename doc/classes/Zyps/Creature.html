new(options = {})

Identical to the GameObject constructor, except that it also takes a list of Behavior objects. Takes a hash with these keys and defaults: 

      :behaviors => []

[Source]

        def initialize (options = {})
                options = {
                        :behaviors => []
                }.merge(options)
                super
                @behaviors = []
                options[:behaviors].each {|behavior| self.add_behavior(behavior)}
        end

        def copy
                copy = super
                #Make deep copy of each behavior.
                copy.instance_eval {@behaviors = []}
                self.behaviors.each {|behavior| copy.add_behavior(behavior.copy)}
                copy
        end
