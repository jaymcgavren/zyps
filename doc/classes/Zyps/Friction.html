        def act(environment)
                elapsed_time = @clock.elapsed_time
                environment.objects.each do |object|
                        #Slow object.
                        acceleration = @force * elapsed_time
                        speed = object.vector.speed
                        if speed > 0
                                speed -= acceleration 
                                speed = 0 if speed < 0
                        elsif speed < 0
                                speed += acceleration
                                speed = 0 if speed > 0
                        end
                        object.vector.speed = speed
                end
        end
